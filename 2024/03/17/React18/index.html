有些改变

我们在引入ReactDOM时候
import ReactDOM from 'react-dom/client'
//这次应该引入的式客户端，这样控制台就不会报错

这个改完以后

之前的这样的熏染就使用不了，

const div = React.createElement('div',[],'我是王昌晨')
//用于创建react根元素
ReactDOM.render(<div></div>,document.getElementById('root'))

需要向下面这样写，这样次啊能成功熏染

import ReactDOM from 'react-dom/client'

const div = React.createElement('div',[],'我是王昌晨')
//获取React根元素对象
const root = ReactDOM.createRoot(document.getElementById('root'))
//使用render进行熏染
root.render(div)

函数式组件的渲染的过程

import React, {Component} from 'react';
import ReactDOM from 'react-dom/client'

function App(){
    return (
        <div>函数式组件</div>
    )
}

//获取React根元素对象
const root = ReactDOM.createRoot(document.getElementById('root'))
//使用render进行熏染
root.render(<App></App>)
